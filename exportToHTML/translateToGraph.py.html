3    # Takes in CSV file of disease hierarchy 
4    #   and writes out json file of disease hierarchy. 
5    # 
6    # Translates this: 
7    #   Acne,67548,Derm 
8    #   Anemia,160481,Blood 
9    # 
10   # To this: 
11   # { 
12   #  "name": "Derm", 
13   #  "children": [ 
14   #   {"name": "Acne", "size": 67548} 
15   #  ] 
16   # }, 
17   # { 
18   #  "name": "Blood", 
19   #  "children": [ 
20   #   {"name": "Anemia", "size": 160481} 
21   #  ] 
22   # },

24   import json 
25   from operator import itemgetter, attrgetter 
26    
27   lookup = {} 
28   emptySet = [] 
29    
30   with open('disease_node_hierarchy_and_size_fv.csv', 'r') as f: 
31       for line in f: 
32           #print line 
33           line = line.rstrip() 
34           a = line.split(',') 
35           a[0] = a[0].replace('"', '') 
36           if a[2]=='': 
37               a[2]='No category' 
38           if a[2] in lookup: 
39               lookup[a[2]].append( [a[0], a[1]] ) 
40           else: 
41               lookup[a[2]] = [ [a[0], a[1]] ] 
42           #print "creating new class:", a[2] 
43    
44   comorbid = {} 
45   lineNumber = 0 
46   with open('comorbid_fv.csv', 'r') as f: 
47       for line in f: 
48           lineNumber+=1 
49           if (lineNumber!=1): 
50               line = line.rstrip() 
51               a = line.split(',') 
52               source = a[0].replace('"', '') 
53               target = a[1].replace('"', '') 
54               comorbidCount = int(a[2]) 
55               if a[0] in comorbid: 
56                   comorbid[source].append( [comorbidCount, target] ) 
57               else: 
58                   comorbid[source] = [[comorbidCount, target]] 
59    
60   drugs = {} 
61   lineNumber = 0 
62   with open ('drugs.csv', 'r') as f: 
63       for line in f: 
64           lineNumber+=1 
65       if (lineNumber!=1): 
66    
67           line = line.rstrip() 
68           a = line.split(',') 
69           source = a[0].replace('"','') 
70           target = a[1].replace('"','') 
71           drugsCount = int(a[2]) 
72           if a[0] in drugs: 
73               drugs[source].append( [drugsCount, target] ) 
74           else: 
75               drugs[source] = [[drugsCount, target]] 
76    
77   with open('disease_tree.json', 'w') as f: 
78       f.write('{\n \"name\": \"diseases\",\n \"children\": [\n') 
79       count = 0 
80       for k in lookup.keys(): 
81           count+=1 
82           print "key: ", k 
83           f.write('{\n') 
84           f.write(' \"name\": \"' + k + '\"' + ',\n') 
85           f.write(' \"children\": [\n') 
86           countInner = 0 
87           for i in lookup[k]: 
88               countInner+=1 
89               f.write('    {\"name\": \"' + i[0] + '\", \"size\": ' + i[1] + ',') 
90    
91               # comorbidities... 
92               f.write('  "comorbid": ['); 
93               countInner2 = 0 
94               for c in sorted(comorbid[i[0]], reverse=True): 
95                   countInner2+=1 
96                   f.write('    {"name": "' + c[1] + '", "weight": ' + str(c[0]) + '}') 
97                   if countInner2 != len(comorbid[i[0]]): 
98                       f.write(',') 
99               f.write(']') 
100   
101              # drugs... 
102              f.write('  "drugs": ['); 
103              countInner3 = 0 
104              for c in sorted(drugs[i[0]], reverse=True): 
105                  countInner3+=1 
106                  f.write('    {"drug": "' + c[1] + '", "pct": ' + str(c[0]) + '}') 
107                  if countInner3 != len(drugs[i[0]]): 
108                      f.write(',') 
109              f.write(']') 
110   
111              f.write ('}') 
112              if countInner != len(lookup[k]): 
113                  f.write(',') 
114              f.write('\n') 
115          f.write('  ]\n') 
116          f.write('}') 
117          if count != len(lookup): 
118              f.write(',') 
119          f.write('\n') 
120      f.write(']\n}')
